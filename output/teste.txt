funcao bubble_sort_multidimensional(arr):
    n = comprimento(arr)
    para i em range(n):
        para j em range(0, n-i-1):
            se arr[j][0] > arr[j+1][0]:
                arr[j], arr[j+1] = arr[j+1], arr[j]

# Exemplo de uso
dados = [(3, 'Maria'), (1, 'João'), (2, 'Ana'), (4, 'Pedro')]
escrever("Dados não ordenados:")
para item em dados:
    escrever(item)

bubble_sort_multidimensional(dados)

escrever("\nDados ordenados com base no primeiro elemento de cada tupla:")
para item em dados:
    escrever(item)
